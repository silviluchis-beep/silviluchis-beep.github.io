""" In simple terms, this routine does 5 main things:
1.	Gets metadata from a PXWeb API table (dimensions, values).
2.	Builds a query automatically using those dimensions.
3.	Requests the full dataset from the API.
4.	Converts the result into a Pandas DataFrame.
"""

def fetch_pxweb_to_excel(
    url,
    file_name=None,
    query_type="all",
    preview=True
):

    print(f"\nFetching metadata from: {url}")

    # 1️⃣ Get metadata from PXWeb API
    try:
        meta_response = requests.get(url, timeout=30)  # request metadata
        meta_response.raise_for_status()  # raise error if request failed
        meta = meta_response.json()  # convert JSON response to Python dictionary
    except requests.exceptions.RequestException as e:
        print(f" Error fetching metadata: {e}")
        return None

    # Show dataset dimensions
    print("Dimensions found:")
    for var in meta["variables"]:
        print(f"  {var['code']} → {len(var['values'])} values")

    # 2️⃣ Build the API query automatically
    if query_type == "all":

        # Select all values using wildcard "*"
        query = {
            "query": [
                {
                    "code": var["code"],
                    "selection": {
                        "filter": "all",
                        "values": ["*"]
                    }
                }
                for var in meta["variables"]
            ],
            "response": {"format": "json-stat"}
        }

    elif query_type == "item":

        # Select all values explicitly
        query = {
            "query": [
                {
                    "code": var["code"],
                    "selection": {
                        "filter": "item",
                        "values": var["values"]
                    }
                }
                for var in meta["variables"]
            ],
            "response": {"format": "json-stat"}
        }

    else:
        raise ValueError("query_type must be 'all' or 'item'")

    # 3️⃣ Send POST request to fetch dataset
    headers = {
        "Content-Type": "application/json",
        "Accept": "application/json",
        "User-Agent": "Mozilla/5.0"
    }

    try:
        response = requests.post(
            url,
            json=query,
            headers=headers,
            timeout=60
        )

        if response.status_code != 200:
            print(" API returned an error:")
            print(response.text)

        response.raise_for_status()

    except requests.exceptions.RequestException as e:
        print(f"Error fetching data: {e}")
        return None

    # 4️⃣ Convert API response to pandas DataFrame
    dataset = pyjstat.Dataset.read(response.text)  # read JSON-stat
    df = dataset.write("dataframe")  # convert to DataFrame

    # 5️⃣ Optional preview of dataset
    if preview:
        print("\nDataset preview:")
        print(df.head())

        print(f"\nDataset shape: {df.shape}")
 

    return df

Fetch quarterly Finnish GDP from the PXWeb API,
convert the dataset to a Pandas DataFrame,
and save it to an Excel file.
"""

def fetch_gdp_only(url, file_name="finland_gdp_quarterly"):

    # Print the URL being accessed (useful for tracking which dataset is being processed)
    print(f"\nFetching metadata from: {url}")

    # 1️⃣  Retrieve metadata from the PXWeb API
    # Metadata contains information about dataset dimensions and available values
    meta_response = requests.get(url)
    meta_response.raise_for_status()   # Raise an error if the request fails
    meta = meta_response.json()        # Convert response into a Python dictionary

    # 2️⃣ Build the query to request the desired data from the API
    # Here we specify exactly which variables we want

    query = {
    "query": [
        # Select the economic variable (Taloustoimi)
        # "B1GMH" corresponds to Gross Domestic Product (GDP)
        {
            "code": "Taloustoimi",
            "selection": {
                "filter": "item",
                "values": ["B1GMH"]
            }
        },

        # Select the type of information (Tiedot)
        # "kausitvv2015" refers to seasonally adjusted volume data
        {
            "code": "Tiedot",
            "selection": {
                "filter": "item",
                "values": ["kausitvv2015"]
            }
        },
        
        # Select the time dimension (quarters)
        # "filter": "all" with "*" means retrieve all available quarters
        {
            "code": "Vuosineljännes",
            "selection": {
                "filter": "all",
                "values": ["*"]
            }
        }
    ],

    # Request the response in JSON-stat format
    "response": {"format": "json-stat"}
}

    # 3️⃣ Fetch the dataset from the API
    headers = {
        "Content-Type": "application/json",  # Query is sent as JSON
        "Accept": "application/json"         # Expect JSON response
    }
   
    # Send POST request with the query
    response = requests.post(url, json=query, headers=headers)

    #  If an error occurs, print the API response message for debugging
    if response.status_code != 200:
        print(response.text)

    # Raise an exception if the request failed
    response.raise_for_status()

    # 4️⃣ Convert to DataFrame
    dataset = pyjstat.Dataset.read(response.text)  # Parse JSON-stat dataset
    df = dataset.write("dataframe")          # Convert dataset to DataFrame


    return df

"""
    Fetch Employment and Hours Worked quarterly (1995-2025) 
    and filter to E1, E2, seasonally adjusted series.
"""
def fetch_employment_hours(url):
    # Define HTTP headers for the API request
    headers = {
        "Content-Type": "application/json",  # send query as JSON
        "Accept": "application/json",  # expect JSON response
        "User-Agent": "Mozilla/5.0"    # mimic a browser request
    }

    # 1️⃣ Get metadata from the PXWeb API
    # Metadata contains information about the dataset dimensions and values
    meta = requests.get(url, headers=headers).json()
    # This list will store the query definition for each dataset dimension
    query_parts = []

    # Loop through all variables (dimensions) in the dataset
    for var in meta["variables"]:
        values = var["values"]  # all possible values for this dimension

        # Filter years for time dimension (quarters)
        if var["code"] == "Vuosineljännes":
            selected_values = [v for v in values if 1995 <= int(v[:4]) <= 2025]
            print(f"Limiting {var['code']} to years 1995-2025")
        # Filter Taloustoimi to E1 and E2
        elif var["code"] == "Taloustoimi":
            selected_values = ["E1", "E2"]
            print(f"Filtering {var['code']} to E1 and E2")
        # Filter Tiedot to seasonally adjusted
        elif var["code"] == "Tiedot":
            selected_values = ["KAUSIT"]
            print(f"Filtering {var['code']} to seasonally adjusted (KAUSIT)")
        else:
            selected_values = values

        query_parts.append({
            "code": var["code"],
            "selection": {
                "filter": "item",
                "values": selected_values
            }
        })

    query = {
        "query": query_parts,
        "response": {"format": "json-stat"}
    }

    # 2️⃣ Fetch data from the PXWeb API
    response = requests.post(url, json=query, headers=headers)
    # Raise an error if the API request fails
    response.raise_for_status()

    # 3️⃣ Convert the JSON-stat response into a pandas DataFrame
    dataset = pyjstat.Dataset.read(response.text)  # parse JSON-stat
    df = dataset.write("dataframe")                # convert to DataFrame

    return df

"""
    Fetch official inflation (annual change %) from
    CPI Annual Changes table (122p) for months 2009-2025,
    convert to quarterly averages, and skip older months automatically.
"""

def fetch_inflation_1995_2025(url):

    headers = {
        "Content-Type": "application/json",
        "Accept": "application/json",
        "User-Agent": "Mozilla/5.0"
    }

    # 1️⃣ Get metadata
    meta = requests.get(url, headers=headers).json()
    query_parts = []

    for var in meta["variables"]:
        code = var["code"]
        values = var["values"]

        if code.lower().startswith("month"):
            # Select only months 2009–2025
            selected = [v for v in values if 1995 <= int(v[:4]) <= 2025]
            print(f"✔ Limiting {code} to 1995-2025")
        else:
            selected = values

        query_parts.append({
            "code": code,
            "selection": {
                "filter": "item",
                "values": selected
            }
        })

    query = {
        "query": query_parts,
        "response": {"format": "json-stat"}
    }

    # 2️⃣ Fetch data
    response = requests.post(url, json=query, headers=headers)
    response.raise_for_status()

    # 3️⃣ Convert to DataFrame
    dataset = pyjstat.Dataset.read(response.text)
    df = dataset.write("dataframe")

    # 4️⃣ Rename column
    df = df.rename(columns={"value": "inflation_yoy_pct"})

    # 5️⃣ Convert month strings to datetime
    # PXWeb month format: '2009M01' -> '2009-01'
    df["Month"] = df["Month"].str.replace("M", "-")
    # Convert to datetime
    df["Month"] = pd.to_datetime(df["Month"], format="%Y-%m")
    #*****
    # Filter 2009–2025
    df = df[(df["Month"].dt.year >= 1995) & (df["Month"].dt.year <= 2025)]
    #*****

    # 6️⃣ Convert to quarterly
    df["Quarter"] = df["Month"].dt.to_period("Q")

    # 7️⃣ Aggregate quarterly averages
    df_q = df.groupby("Quarter")["inflation_yoy_pct"].mean().reset_index()

    return df_q

def fetch_ttvi_1995_2025_split(url, folder="Excel_files", file_name="Volume Index Industrial"):
    """
    Fetch Volume Index Industrial Output data safely from 1995–2025 by splitting the request
    into smaller year ranges to avoid API 403 errors.
    """
    # Define the year ranges
    year_ranges = [(1995, 2000), (2001, 2005), (2006, 2010),
                   (2011, 2015), (2016, 2020), (2021, 2025)]

    # HTTP headers
    headers = {
        "Content-Type": "application/json",
        "Accept": "application/json",
        "User-Agent": "Mozilla/5.0"
    }

    # Request metadata once
    meta = requests.get(url).json()

    all_chunks = []

    # Loop over each year range
    for start_year, end_year in year_ranges:
        query_parts = []
        for var in meta["variables"]:
            values = var["values"]

            # Detect time dimension (assume long dimension >100 values)
            if len(values) > 100:
                selected_values = [v for v in values if start_year <= int(v[:4]) <= end_year]
                print(f"Fetching {var['code']} for years {start_year}-{end_year}")
            else:
                selected_values = values

            query_parts.append({
                "code": var["code"],
                "selection": {
                    "filter": "item",
                    "values": selected_values
                }
            })

        # Build query
        query = {
            "query": query_parts,
            "response": {"format": "json-stat"}
        }

        # Send POST request
        response = requests.post(url, json=query, headers=headers)
        response.raise_for_status()

        # Convert JSON-stat to DataFrame
        dataset = pyjstat.Dataset.read(response.text)
        df_chunk = dataset.write("dataframe")
        all_chunks.append(df_chunk)

    # Combine all chunks
    df_full = pd.concat(all_chunks, ignore_index=True)

    return df_full

def perform_eda(df, df_name="Dataset"):
    """
    Performs basic exploratory data analysis on a given dataframe.
    Prints summaries, missing values, correlations, and visualizations.
    """
    
    print(f"\n--- EDA Report for {df_name} ---\n")
    
    # 1. Shape and info
    print("Shape of DataFrame:", df.shape)
    print("\nData Types and Info:")
    print(df.info())
    
    # 2. Display first few rows
    print("\nFirst 5 Rows:")
    print(df.head())
    
    # 3. Summary statistics
    print("\nSummary Statistics:")
    print(df.describe(include='all'))
    
    # 4. Missing values
    print("\nMissing Values:")
    print(df.isnull().sum())
    
    # 5. Correlation heatmap (only numeric columns)
    numeric_cols = df.select_dtypes(include=np.number).columns
    if len(numeric_cols) > 1:
        plt.figure(figsize=(10,6))
        sns.heatmap(df[numeric_cols].corr(), annot=True, fmt=".2f", cmap="coolwarm")
        plt.title(f"Correlation Matrix - {df_name}")
        plt.show()
    
    # 6. Histograms for numeric columns
    for col in numeric_cols:
        plt.figure(figsize=(6,3))
        sns.histplot(df[col], kde=True, bins=20)
        plt.title(f"Distribution of {col} - {df_name}")
        plt.show()
    
    
    print(f"\n--- End of EDA for {df_name} ---\n")

def save_to_excel(df, output_file_name):
    """
    Saves a DataFrame to an Excel file inside the 'Excel_files' folder.
    
    Parameters:
    df : pandas DataFrame
        The DataFrame to save.
    output_file_name : str
        The name of the output Excel file.
    """

    # Define the folder where files will be saved
    folder = "Excel_files"

    # Create the folder if it does not exist
    os.makedirs(folder, exist_ok=True)

    # Build the full file path
    output_file = os.path.join(folder, output_file_name)

    # Save the DataFrame to Excel
    df.to_excel(output_file, index=False)

    # Confirmation message
    print(f"✅ File saved: {output_file}")

"""
    Cleans a dataframe downloaded from PX-Web (Statistics Finland).
    Steps included:
    1. Replaces missing value markers (“..”) with pandas NA
    2. Converts numeric columns to proper numeric type when possible
    3. Drops fully empty rows
    4. Resets the dataframe index
    
    Returns:
        A cleaned copy of the original dataframe ready for analysis.
    """

def clean_pxweb_dataframe(df):
     # Make a copy of the original dataframe to avoid modifying it directly
    df = df.copy()
    
    # 1️⃣ Replace missing markers
    # PX-Web datasets often use ".." to indicate missing values; convert these to pandas NA
    df.replace("..", pd.NA, inplace=True)
    
    # 2️⃣ Convert numeric columns when possible
    # Attempt to convert each column to numeric if possible; non-numeric columns remain unchanged
    for col in df.columns:
        try:
            df[col] = pd.to_numeric(df[col])
        except:
            # If conversion fails (e.g., for string columns), just skip it
            pass
    
    # 3️⃣ Drop rows where all values are missing
    df.dropna(how="all", inplace=True)
    
    # 4️⃣ Reset the index after dropping rows
    df.reset_index(drop=True, inplace=True)
    
    return df

"""
This routine creates tables in the SQLite database for storing population, employment, and region data. If the tables already exist, it does nothing. 
It ensures the database is ready to store cleaned and organized data.
"""

def create_population_tables(conn):
    cursor = conn.cursor()

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS population_total (
            year INTEGER,
            age_group TEXT,
            information TEXT,
            value REAL
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS population_males (
            year INTEGER,
            age_group TEXT,
            information TEXT,
            value REAL
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS population_females (
            year INTEGER,
            age_group TEXT,
            information TEXT,
            value REAL
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS population_region (
            year INTEGER,
            information TEXT,
            value REAL,
            region_code TEXT
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS region_dim (
            region_code TEXT,
            region_name TEXT
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS employed_industry (
            year INTEGER,
            sex TEXT,
            industry TEXT,
            information TEXT,
            value REAL
        )
    """)


    conn.commit()
    print("✅ Tables created")

""""
This routine splits the population dataset by sex and inserts the data into three separate tables in the SQLite database: Total, Males, and Females.
"""

def insert_split_population_data(conn, df):

    
    # Optional: clear tables to avoid duplicates
    conn.execute("DELETE FROM population_total")
    conn.execute("DELETE FROM population_males")
    conn.execute("DELETE FROM population_females")

    
    df_total = df[df["Sex"] == "Total"].drop(columns=["Sex"])
    df_males = df[df["Sex"] == "Males"].drop(columns=["Sex"])
    df_females = df[df["Sex"] == "Females"].drop(columns=["Sex"])

    df_total.to_sql("population_total", conn, if_exists="append", index=False)
    df_males.to_sql("population_males", conn, if_exists="append", index=False)
    df_females.to_sql("population_females", conn, if_exists="append", index=False)

    print("✅ Data inserted into 3 tables")

"""
This routine inserts the region dimension table and fact tables for population by region and employed by industry into the SQLite database. 
Existing data is cleared first to avoid duplicates.
"""

def insert_population_region_data(conn, df_fact, region_dim, df_fact2):

    # Optional: clear tables to avoid duplicates
    conn.execute("DELETE FROM population_region")
    conn.execute("DELETE FROM region_dim")
    conn.execute("DELETE FROM employed_industry")


    # Insert dimension table
    region_dim.to_sql(
        "region_dim",
        conn,
        if_exists="append",
        index=False
    )


    df_fact.to_sql(
        "population_region",
        conn,
        if_exists="append",
        index=False
    )

    df_fact2.to_sql(
        "employed_industry",
        conn,
        if_exists="append",
        index=False
    )

    print("✅ Data inserted into population_region and region_dim")

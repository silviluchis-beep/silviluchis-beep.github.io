from shapely.geometry import Point
import random

def random_point_in_polygon(polygon):
    minx, miny, maxx, maxy = polygon.bounds
    while True:
        # random lon, lat
        x = random.uniform(minx, maxx)
        y = random.uniform(miny, maxy)
        point = Point(x, y)
        if polygon.contains(point):
            return point

# generate N random points
N = 50
points = [random_point_in_polygon(finland_shape) for _ in range(N)]

# convert to a GeoDataFrame

locations_df = gpd.GeoDataFrame(geometry=points)
locations_df["Longitude"] = locations_df.geometry.x
locations_df["Latitude"] = locations_df.geometry.y

from sklearn.cluster import KMeans

def run_kmeans2(locations_df, k):
    """
    Run KMeans clustering on a dataframe with 'Latitude' and 'Longitude'.
    Returns a new dataframe with cluster labels and the cluster centers.
    """
    
    # Work on a copy
    df_copy = locations_df.copy()

    # Safely remove old cluster column if present
    df_copy = df_copy.drop(columns=["cluster"], errors="ignore")

    # Fit model
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans.fit(df_copy[['Latitude', 'Longitude']])

    # Add clusters
    df_copy['cluster'] = kmeans.labels_

    return df_copy, kmeans.cluster_centers_

from sklearn.cluster import KMeans
import numpy as np

def run_kmeans_basic(locations_df, k):
    """
    Runs KMeans clustering on the given dataframe using Longitude and Latitude.
    
    Parameters:
        locations_df (pd.DataFrame): DataFrame with 'Longitude' and 'Latitude' columns.
        k (int): Number of clusters.
    
    Returns:
        X (ndarray): Original coordinate array.
        cluster_pred (ndarray): Cluster labels for each point.
        cluster_center (ndarray): Coordinates of cluster centers.
    """
    
    # Extract coordinates as a NumPy array
    X = locations_df[["Longitude", "Latitude"]].values
    
    # Run KMeans
    kmeans1 = KMeans(n_clusters=k, random_state=42)
    kmeans1.fit(X)
    
    # Predictions and cluster centers
    cluster_pred = kmeans1.predict(X)
    cluster_center = kmeans1.cluster_centers_
    
    return X, cluster_pred, cluster_center

import pandas as pd

def create_centroids_df(locations_df, cluster_centers):
    """
    Creates a DataFrame for cluster centers using the same column names
    as the original coordinate columns.

    Parameters:
        locations_df (pd.DataFrame): Original dataframe with 'Longitude' and 'Latitude'.
        cluster_centers (array-like): Numpy array of cluster center coordinates.

    Returns:
        pd.DataFrame: DataFrame containing the cluster centers.
    """
    
    # Extract the column names used for coordinates
    colnames = locations_df[["Longitude", "Latitude"]].columns
    
    # Create the DataFrame
    centroids_df = pd.DataFrame(cluster_centers, columns=colnames)
    
    return centroids_df

import matplotlib.pyplot as plt
import geopandas as gpd
from shapely.geometry import Point
import seaborn as sns

def plot_finland_clusters2(locations_df, cluster_centers, world_df):
    """
    Plots the Finland outline, the clustered points, 
    and the cluster centers with colorful visualization.
    """
    
    # Extract Finland polygon
    finland = world_df[world_df['ADMIN'] == 'Finland']
    
    # Convert to GeoDataFrame
    points = gpd.GeoDataFrame(
        locations_df,
        geometry=[Point(xy) for xy in zip(locations_df["Longitude"], locations_df["Latitude"])],
        crs="EPSG:4326"
    )

    # Create plot
    fig, ax = plt.subplots(figsize=(12, 14))

    # --- Plot Finland outline ---
    finland.boundary.plot(ax=ax, edgecolor='black', linewidth=1.2)

    # --- Colorful cluster plotting ---
    unique_clusters = sorted(points["cluster"].unique())
    palette = sns.color_palette("tab10", len(unique_clusters))

    for cluster_id, color in zip(unique_clusters, palette):
        cluster_points = points[points["cluster"] == cluster_id]
        cluster_points.plot(
            ax=ax,
            color=color,
            markersize=20,
            label=f"Cluster {cluster_id}"
        )

    # --- Cluster centers with nice styling ---
    plt.scatter(
        cluster_centers[:, 0],
        cluster_centers[:, 1],
        s=250,
        c="yellow",
        edgecolors="black",
        linewidth=1.5,
        marker='X',
        label="Cluster Centers"
    )
    
    # --- Zoom around Finland ---
    minx, miny, maxx, maxy = finland.total_bounds
    ax.set_xlim(minx - 2, maxx + 2)
    ax.set_ylim(miny - 2, maxy + 2)

    # Titles & labels
    ax.set_title("Clustered Delivery Points in Finland", fontsize=16)
    ax.set_xlabel("Longitude", fontsize=12)
    ax.set_ylabel("Latitude", fontsize=12)

    ax.legend(title="Legend")
    
    plt.show()

def run_clustering(model, data):
    start = default_timer()
    labels = model.fit_predict(data)
    end = default_timer()
    elapsed = end - start
    return labels, elapsed

def compare_kmeans_vs_agglomerative(data, k):
    
    print(f"\n=== Comparing KMeans and Agglomerative for k={k} ===")
    
    # --- Run KMeans ---
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans_labels, kmeans_time = run_clustering(kmeans, data)
    
    # --- Run Agglomerative ---
    agg = AgglomerativeClustering(n_clusters=k)
    agg_labels, agg_time = run_clustering(agg, data)
    
    # --- Visualization ---
    fig, ax = plt.subplots(1, 2, figsize=(14, 6))
    
    sns.scatterplot(
        x=data[:,0], y=data[:,1],
        hue=kmeans_labels,
        palette="tab10",
        ax=ax[0]
    )
    ax[0].set_title(f"KMeans Clustering (k={k})")
    ax[0].legend().remove()
    
    sns.scatterplot(
        x=data[:,0], y=data[:,1],
        hue=agg_labels,
        palette="tab10",
        ax=ax[1]
    )
    ax[1].set_title(f"Agglomerative Clustering (k={k})")
    ax[1].legend().remove()
    
    plt.show()
    
    # --- Print timing ---
    print(f"KMeans time:          {kmeans_time:.6f} seconds")
    print(f"Agglomerative time:   {agg_time:.6f} seconds")
    
    # --- Conclusions (automatic) ---
    print("\n--- Automatic Observations ---")
    
    if kmeans_time < agg_time:
        print("* KMeans was faster.")
    else:
        print("* Agglomerative was faster.")
    
    if len(np.unique(kmeans_labels)) != len(np.unique(agg_labels)):
        print("* The methods produced a different number of clusters (unexpected).")
    
    print("* You can visually compare how compact / spread out clusters are in the plots.")

pclass = ["First","Second","Third"]


# Plot survival by embark group

for i in (range(0,len(labels))):
    for j in (range(0,len(pclass))):
        sns.countplot(data=(df[(df["parch"] > 0) & (df["embarked"]== labels[i]) & (df["sibsp"]>=1) & (df["class"]==pclass[j])]),x='who',hue='alive', palette='pastel')
